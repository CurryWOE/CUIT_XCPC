## 区间更新、单点查询
A为原数组，D为差分数组

$A[i] = \sum\limits_{j=1}^i D[j]$

当 $[l,r]$ 值改变了，只有 D[l] 和 D[r+1] 的值发生改变
```c++
struct BIT
{
    uint32_t n;
    int64_t a[N];
    void init(uint32_t n)
    {
        this->n=n;
        memset(a,0,sizeof(a[0])*(n+1));
    }

            a[pos]+=val;

        update(r+1,-val);

            res+=a[pos];
        return res;

}bit;
```

## 区间更新、区间查询
A为原数组，D为差分数组

$\sum\limits_{i=1}^n A[i] = \sum\limits_{i=1}^n \sum\limits_{j=1}^i D[j]$

$=\sum\limits_{i=1}^n (n+1-i)\times D[i]$

$=\left[n\sum\limits_{i=1}^n D[i]\right]-\left[\sum\limits_{i=1}^n (i-1)D[i]\right]$

维护两个树状数组，sum1[i] = D[i]，sum2[i] = D[i]*(i-1);
```c++
struct BIT
{
    uint32_t n;
    int64_t a[N],b[N];
    void init(uint32_t n)
    {
        this->n=n;
        memset(a,0,sizeof(a[0])*(n+1));
        memset(b,0,sizeof(b[0])*(n+1));
    }
    void update(uint32_t pos,int64_t val)
    {
        for(auto x=pos;pos<=n;pos+=pos&-pos)
        {
            a[pos]+=val;
            b[pos]+=val*(x-1);
        }
    }
    void update(uint32_t l,uint32_t r,int64_t val)
    {
        update(l,val);
        update(r+1,-val);
    }
    int64_t query(uint32_t pos)
    {
        int64_t res=0;
        for(auto x=pos;pos;pos&=pos-1)
            res+=x*a[pos]-b[pos];
        return res;
    }
    int64_t query(uint32_t l,uint32_t r)
    {
        return query(r)-query(l-1);
    }
}bit;
bit.init();
for(uint32_t i=1;i<=n;++i)
    bit.update(i,a[i] - a[i-1]);
bit.update(l,r,k);
bit.query(l,r);
```

## 二维树状数组
```cpp
struct BIT
{
    uint32_t n,m;
    int32_t sum[N][N];
    void init(uint32_t nn,uint32_t mm)
    {
        n=nn;
        m=mm;
        for (uint32_t i = 1; i <= n; ++i)
            for (uint32_t j = 1; j <= m; ++j)
                sum[i][j] = 0;
    }
    void update(uint32_t x, uint32_t y, int32_t val)
    {
        auto tmp = y;
        for (; x <= n; x+=x&-x)
            for (y = tmp; y <= m; y+=y&(-y))
                sum[x][y] += val;
    }
    int32_t query(uint32_t x, uint32_t y)
    {
        auto tmp=y;
        int32_t res = 0;
        for (; x > 0; x&=x-1)
            for (y = tmp; y > 0; y&=y-1)
                res += sum[x][y];
        return res;
    }
    int32_t query(uint32_t x1, uint32_t y1, uint32_t x2, uint32_t y2)
    {
        return query(x2, y2) - query(x1 - 1, y2) - query(x2, y1 - 1) + query(x1 - 1, y1 - 1);
    }
} bit;
```
