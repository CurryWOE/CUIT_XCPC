# 线段树进阶技巧
# 线段树上二分
注意到线段树是一颗二叉树，如果信息满足二分性，那么线段树天然适合二分

这里以求全局第k小为例，如果不使用排序，可以用权值线段树+线段树上二分解决

至于为什么不使用排序，因为这个技巧是区间第k小的前置知识，直接学区间第k小，难度跨度有点大

而且不止这种信息可以用线段树上二分，[例题](https://codeforces.com/contest/1741/problem/F)

---
设当前节点表示值域 $[l,r]$，想求这个值域的第k小

设该节点左儿子表示值域 $[l,mid]$ ，左儿子的值域内的数有x个，若x小于k，那说明答案在右儿子的值域内，否则在左儿子的值域内

每次递归把答案所在值域缩小一半，完全符合二分
```cpp
struct SegmentTree
{
    struct node
    {
        int l,r,sum;//这里的l，r表示左右儿子分配到的下标
    }st[N<<2];
    int tot;
    //初始化之后把原序列所有值扔到线段树上
    void update(int &id, int l, int r,int val)
    {
        if(!id)
            id=++tot;
        ++st[id].sum;
        if(l==r)
            return;
        int mid = (l+r)>>1;
        if (val <= mid)
            update(st[id].l,l,mid,val);
        else
            update(st[id].r,mid+1,r,val);
    }
    int query(int id,int segl,int segr,int k)
    {
        if (segl==segr)
            return segl;
        int mid = (segl + segr)>>1;
        if (st[st[id].l].sum>=k)
            return query(st[id].l,segl,mid,k);
        return query(st[id].r,mid+1,segr,k);
    }
    void init()
    {
        for(;tot;--tot)
            st[tot]={0,0,0};
    }
}st;
st.init();
for(int i=1;i<=n;++i)
    st.update(1,MinVal,MaxVal,a[i]);
st.query(1,MinVal,MaxVal,k);
```
# 可持久化(persistent)
笔者感觉这个翻译不够好，但是大家都是这么叫的

定义：初始有一个信息为空的东西，不断对其修改，会产生多个版本，保存每个历史版本就是可持久化

比如photoshop上画画，可以撤销上一次画的东西，也就是回到上一个版本

应用到线段树上，最朴素的方法是，每次修改都新建一颗树，继承上一棵树的信息，同时把这一次修改的信息添加上

但是很明显时空复杂度都不可接受。注意到每次修改只有部分点受到影响，因此可以共用没受到影响的点
```c++
struct persistentSegmentTree
{
    int tot,n,rtnum;//rtnum表示版本数量
    int rt[N];//每棵树根分配到的下标
    struct node
    {
        int l,r;
        ll sum;
    }st[N<<5];
    void build(int &id,int l,int r)
    {
        id=++tot;
        if(l==r)
        {
            st[id].sum=a[l];
            return;
        }
        int mid=(l+r)>>1;
        build(st[id].l,l,mid);
        build(st[id].r,mid+1,r);
        st[id].sum=st[st[id].l].sum+st[st[id].r].sum;
    }
    //单点修改
    void update(int &id,int pre,int l,int r,int pos,ll val)
    {
        id=++tot;
        st[id]=st[pre];//继承上一棵树表示相同区间的对应节点的信息
        st[id].sum+=val;
        if(l==r)
            return;
        int mid=(l+r)>>1;
        if(pos<=mid)
            update(st[id].l,st[id].l,l,mid,pos,val);
        else
            update(st[id].r,st[id].r,mid+1,r,pos,val);
    }
    ll query(int id,int segl,int segr,int l,int r)
    {
        if(!id)
            return 0;
        if(l<=segl && segr<=r)
            return st[id].sum;
        int mid=(segl+segr)>>1;
        ll res=0;
        if(l<=mid)
            res+=query(st[id].l,segl,mid,l,r);
        if(r>mid)
            res+=query(st[id].r,mid+1,segr,l,r);
        return res;
    }
    void update(int root,int pos,ll val)//指定版本的单点修改
    {
        update(rt[++rtnum],rt[root],1,n,pos,val);
    }
    ll query(int root,int l,int r)//指定版本的区间查询
    {
        return query(rt[root],1,n,l,r);
    }
    void init(int n)
    {
        tot=rtnum=0;
        this->n=n;
        build(rt[0],1,n);
    }
}pst;
pst.init(n);
pst.update(rt,pos,val);
pst.query(rt,l,r);
```

# 主席树(可持久化权值线段树+线段树上二分)
区间第k小，可以转化为两个版本相减之后，第k小。版本相减类似于前缀和吧（

求区间前k大，需要特殊处理叶子
```cpp
struct persistentWeightSegmentTree
{
    uint32_t tot,q,rt[N];
    flat_set<int> discretization;
    struct node { int l,r,sum; }st[N<<5];
    void init(int n)//原数组a(1-indexed)
    {
        tot=0;
        discretization = flat_set(a + 1, a + n + 1);
        q = discretization.size();
        for(int i=1;i<=n;++i)
            modify(rt[i],rt[i-1],1,q,discretization.find(a[i]) - discretization.begin() + 1);
    }
    void modify(int &id,int pre,int l,int r,int pos)
    {
        id=++tot;
        st[id]=st[pre];
        ++st[id].sum;
        if(l==r)
            return;
        int mid=(l+r)>>1;
        if(pos<=mid)
            modify(st[id].l,st[id].l,l,mid,pos);
        else
            modify(st[id].r,st[id].r,mid+1,r,pos);
    }
    int query(int u,int v,int l,int r,int k)//查询区间[u,v]的第k小值的下标
    {
        if(l==r)
            return l;
        int mid=(l+r)>>1,x=st[st[v].l].sum-st[st[u].l].sum;
        if(x>=k)
            return query(st[u].l,st[v].l,l,mid,k);
        return query(st[u].r,st[v].r,mid+1,r,k-x);
    }
    int ask(int l,int r,int k)
    {
        return discretization.begin()[query(rt[l - 1], rt[r], 1, q, k) - 1];
    }
}wt;
wt.init(n);
wt.ask(l,r,k);
```
# 扫描线
离线算法，枚举一维，修改查询另一维

以求平面矩形面积的并为例

使用一条垂直于X轴的直线，从左到右来扫描这个图形

明显，只有在碰到矩形的左边界或者右边界的时候，这个线段所扫描到的情况才会改变

所以把所有矩形的入边，出边按X值排序。然后根据X值从小到大去处理，就可以用线段树来维护y方向上哪些点被矩形覆盖了

扫描线扫描到的情况变化时，对图像切一刀，那么原图像就变成多个互不重叠矩形，相邻两个边差值表示宽，y方向上被覆盖长度表示高，高宽相乘得矩形面积

例程是 [hdu1542](http://acm.hdu.edu.cn/showproblem.php?pid=1542)
```cpp
flat_set<double> discreY;
namespace segmentTree
{
    struct node
    {
        int lz;
        double sum;
    } st[M];
    void pushup(int id,int segl,int segr)
    {
        if(st[id].lz)
            st[id].sum = discreY.begin()[segr] - discreY.begin()[segl];
        else
            st[id].sum=st[id<<1].sum+st[id<<1|1].sum;
    }
    void build(int id,int l,int r)
    {
        st[id]={0,0};
        if(r-l<=1)
            return;
        int mid=(l+r)>>1;
        build(id<<1,l,mid);
        build(id<<1|1,mid,r);
    }
    void upd(int id,int segl,int segr,int l,int r,int val)
    {
        if(l<=segl && segr<=r)
        {
            st[id].lz+=val;
            pushup(id,segl,segr);
            return;
        }
        int mid=(segl+segr)>>1;
        if(l<mid)
            upd(id<<1,segl,mid,l,r,val);
        if(r>mid)
            upd(id<<1|1,mid,segr,l,r,val);
        pushup(id,segl,segr);
    }
}
double allY[N*2];
struct Line { double x,y1,y2; int flag; } line[N*2];
double solve(uint32_t n,vector<array<double,4>> arr)
{
    for(uint32_t i=0; const auto [x1,y1,x2,y2]:arr)
    {
        line[++i]={x1,y1,y2,1};
        line[++i]={x2,y1,y2,-1};
    }
    ranges::sort(span(line+1,n*2), {}, &Line::x);
    for(uint32_t i=0; const auto [x1,y1,x2,y2]:arr)
    {
        allY[++i]=y1;
        allY[++i]=y2;
    }
    discreY = flat_set(allY+1, allY+n*2+1);
    auto q = discreY.size()-1;
    double ans=0;
    segmentTree::build(1,0,q);
    for(uint32_t i=1;i<=n*2;++i)
    {
        ans+=segmentTree::st[1].sum*(double)(line[i].x-line[i-1].x);
        segmentTree::upd(1,0,q,discreY.find(line[i].y1) - discreY.begin(),discreY.find(line[i].y2) - discreY.begin(),line[i].flag);
    }
    return ans;
}
```
# 势能线段树
懒标记可以节省开销的前提是：节点的区间信息可以根据懒标记来更新；懒标记之间可以快速相互合并

但是有些区间修改操作不能依靠懒标记完成，比如区间开根号，区间位运算，因为这些运算都是依赖于叶子节点的值

以区间开根为例，1再开根还是1，所以修改没有意义，就是操作已经“退化”了

因此可以暴力开根每个叶子节点。当区间最大值都是1，说明这个区间不用递归进去修改，反正值不会变

而一个数不断开根，变成1需要的次数非常少。所以总操作次数可以保证不会太大

而势能线段树就是每个节点维护一个势能标记，就好像你一开始位于高空，每次修改会让你的高度下降，当你落到地面时，再对你修改就已经没有意义了。就是这个操作对你而言已经"退化"了。

对于区间开根，区间最大值就是势能标记。对于区间取模，区间最大值就是势能标记。对于区间变成其因子数，区间最大值就是势能标记

简而言之，势能线段树就是暴力剪枝
# 线段树优化建图
点向区间内每一个点连同权边，线段树可以用 $O(\log n)$ 个点表示一个区间，因此点向区间连边的边数从 $O(n)$ 降到 $O(\log n)$

下面给出两种写法，zkw线段树和递归线段树的速度差不多，zkw线段树代码更短

为了方便点和点连边的写法不变，将前n个点分配给原始点，后面的点分配给线段树上的点。且经测试，这样写速度快，代码短
```cpp
namespace zkwSegmentTreeBuildGrapth
{
    int leafBegin,in[N<<2],out[N<<2];//in表示点向区间，out表示区间向点
    void init(int nn)
    {
        leafBegin=bit_ceil((unsigned int)(nn+2));//#include <bit>
        iota(in,in+leafBegin,nn);//#include <numeric>
        iota(in+leafBegin,in+leafBegin+nn+1,0);
        iota(out,out+leafBegin,nn+leafBegin);
        iota(out+leafBegin,out+leafBegin+nn+1,0);
        for(int i=1;i<leafBegin;++i)
        {
            dijkstra::add(in[i],in[i<<1],0);
            dijkstra::add(in[i],in[i<<1|1],0);
        }
        for(int i=1;i<leafBegin;++i)
        {
            dijkstra::add(out[i<<1],out[i],0);
            dijkstra::add(out[i<<1|1],out[i],0);
        }
    }
    void addEdge(int l,int r,int x,int w)//点向区间
    {
        for(int L=leafBegin+l-1,R=leafBegin+r+1;L^R^1;L>>=1,R>>=1)
        {
            if(~L&1)
                dijkstra::add(x,in[L+1],w);
            if(R&1)
                dijkstra::add(x,in[R-1],w);
        }
    }
    void addEdge2(int l,int r,int x,int w)//区间向点
    {
        for(int L=leafBegin+l-1,R=leafBegin+r+1;L^R^1;L>>=1,R>>=1)
        {
            if(~L&1)
                dijkstra::add(out[L+1],x,w);
            if(R&1)
                dijkstra::add(out[R-1],x,w);
        }
    }
}
```

# 李超树(Li-Chao Tree)
[论文](https://arxiv.org/html/2603.07948)里面讲的很好，这里不抄细节

$O(\log C)$ 插入直线，$O(\log^2 C)$ 插入线段，$O(\log C)$ 查询和x=b这条竖直线相交的线的最值信息

李超树的线段树合并，和一般线段树合并相似，差别：先用__InsertLine插入当前点的直线，再递归合并子节点，不需要pushup函数
```cpp
namespace LiChaoTree
{
    using X=int;
    using Y=double;
    const X X_MIN=1,X_MAX=4e4;
    const int N=1e5+3;
    struct Line
    {
        Y k, m;
        inline Y eval(X x) const { return k * x + m; }
    }line[N];
    int lict[(X_MAX-X_MIN+1)<<2],cnt=0;
    //返回 true 如果 lineA 比 lineB 好 当横坐标等于x
    inline bool better(int lineA,int lineB,X x){}
    //处理斜率不存在的线，应返回的答案
    inline Y verticalValue(Y y0, Y y1){}
    void __InsertLine(int id,X l,X r,int LineID)
    {
        while(true)
        {
            X mid=(l+r)>>1;
            if (better(LineID,lict[id],mid))
                swap(lict[id], LineID);
            if (better(LineID,lict[id],l))
            {
                id<<=1;
                r=mid;
            }
            else if(better(LineID,lict[id],r))
            {
                id=id<<1|1;
                l=mid+1;
            }
            else
                break;
        }
    }
    void __InsertSegment(int id,X l,X r, X qL, X qR, int LineID)
    {
        if (qL <= l && r <= qR)
        {
            __InsertLine(id, l, r, LineID);
            return;
        }
        X mid = (l+r)>>1;
        if(mid>=qL)
            __InsertSegment(id<<1, l, mid, qL, qR, LineID);
        if(mid<qR)
            __InsertSegment(id<<1|1, mid + 1, r, qL, qR, LineID);
    }
    int query(X x)
    {
        int ans=0,i=1;
        for(X l=X_MIN,r=X_MAX;;)
        {
            if(better(lict[i],ans,x))
                ans=lict[i];
            if(l==r)
                break;
            X mid = (l+r)>>1;
            if (x <= mid)
            {
                i=i<<1;
                r=mid;
            }
            else
            {
                i=i<<1|1;
                l=mid+1;
            }
        }
        return ans;
    }
    void addLineOrSegment(X x0,X y0,X x1,X y1,bool addLine)
    {
        if(x0==x1)
            line[++cnt]={0,verticalValue(y0,y1)};
        else
        {
            line[++cnt].k=(Y)(y1-y0)/(Y)(x1-x0);
            line[cnt].m=y0-line[cnt].k*x0;
        }
        if(addLine)
            __InsertLine(1,X_MIN,X_MAX,cnt);
        else
            __InsertSegment(1,X_MIN,X_MAX,min(x0,x1),max(x0,x1),cnt);
    }
}
```
# 线段树合并
合并线段树，意味着多棵线段树，而多棵线段树由于空间通常不够，需要动态开点

具体合并实现，就是在两颗线段树上以相同走法遍历所有点，如果一颗树的某点有信息，而另一个颗树的对应点没有，就返回有信息的那个点作为父节点的儿子，否则合并信息之后再作为儿子

以任意顺序合并多棵线段树的总时间复杂度 $O(合并前总节点数 - 合并后总节点数)$

进一步推论，值域为 N 的多棵线段树进行过 k 次插入操作，合并的时间复杂度为 $O(k\log N - 合并后总节点数) < O(k\log N)$

例程提供了两种写法，一种是覆盖旧信息，一种是新开点存新信息

覆盖旧信息的写法，在连续合并多棵线段树时，如果需要中间过程的线段树的结果，就需要在合并的同时去计算信息，不能全部合并完了再去计算信息，因为旧信息已经被覆盖了

覆盖旧信息的写法，的好处是不会增加点数，省空间
```cpp
namespace SegmentTree
{
    struct node
    {
        int info;
        int l=0,r=0;
    }st[M];
    int root[N],tot=0;
    void update(int &id,int segl,int segr,int pos, int val)
    {
        if(!id)
            id=++tot;
        if (segl==segr)
        {
            //calculate infomation in st[id]
            return;
        }
        int mid = (segl + segr)>>1;
        if (pos <= mid)
            update(st[id].l,segl,mid,pos,val);
        else
            update(st[id].r,mid+1,segr,pos,val);
        pushup(id);
    }
    void merge(int &x,int y,int l,int r)
    {
        if(!x || !y)
        {
            x |= y;
            return;
        }
        if(l==r)
        {
            //merge infomation from st[y] to st[x]
            return;
        }
        int mid=(l+r)>>1;
        merge(st[x].l,st[y].l,l,mid);
        merge(st[x].r,st[y].r,mid+1,r);
        pushup(x);
    }
    int merge(int x,int y,int l,int r)
    {
        if(!x || !y)
            return x+y;
        int id=++tot;
        if(l==r)
        {
            //merge infomation from st[y] and st[x] to st[id]
            return id;
        }
        int mid=(l+r)>>1;
        st[id].l=merge(st[x].l,st[y].l,l,mid);
        st[id].r=merge(st[x].r,st[y].r,mid+1,r);
        pushup(x);
        return id;
    }
}
```

# 猫树
[源博客](https://immortalco.blog.uoj.ac/blog/2102)，线段树的一个节点，不再存储一个区间的信息，而是存储区间中点mid开始的后缀信息，mid+1开始的前缀信息，这样建树的时空复杂度从 $O(n)$ 变成 $O(n\log n)$

线段树上代表区间 [l,l] 的点，线段树上代表区间 [r,r] 的点，两点的LCA，使用该点存储的前后缀信息，单次查询的信息合并次数从普通线段树的 $O(\log n)$ 变成 $O(1)$

为了快速求LCA，满二叉树，且静态开点，节点 l 和 r 的LCA的深度，等于 l^r 的最高位
```cpp
const uint32_t N=1e5+1;
namespace CatTree
{
    int a[N],ct[bit_width(N)][N];
    inline void init(uint32_t n)
    {
        for(uint32_t i=0;i<bit_width(n);++i)
            memcpy(ct[i],a,sizeof(a[0])*n);
        for (uint32_t s = 1; s < bit_width(n); ++s)
        {
            int* c = ct[s];
            for (uint32_t mid = 1<<s; mid < n; mid += (1 << (s + 1)))
            {
                for (uint32_t i = mid - 1, l = mid - (1 << s); i > l; --i)
                    ct[s][i - 1] = min(ct[s][i], ct[s][i - 1]);
                for (uint32_t i = mid, r = min(n, mid + (1 << s)) - 1; i < r; ++i)
                    ct[s][i + 1] = min(ct[s][i], ct[s][i + 1]);
            }
        }
    }
    inline int query(uint32_t l,uint32_t r)
    {
        uint32_t s=(l==r?0:bit_width(l^r)-1);
        return min(ct[s][l],ct[s][r]);
    }
}
```

## 分块猫树
块内暴力，块间猫树，块前后缀。[洛谷RMQ速度第一](https://www.luogu.com.cn/record/276102344)
```cpp
const uint32_t N=1e5+1,K=32;
namespace BlockCatTree
{
    uint32_t a[N],prefixInBlock[N],suffixInBlock[N],ct[bit_width(N/K)][N/K];
    inline void init(uint32_t n)
    {
        memcpy(prefixInBlock,a,sizeof(uint32_t)*n);
        for(uint32_t l=0;l<n;l+=K)
            for(uint32_t i=l+1,r=min(n,l+K);i<r;++i)
                prefixInBlock[i]=max(prefixInBlock[i-1],prefixInBlock[i]);
        memcpy(suffixInBlock,a,sizeof(uint32_t)*n);
        for(uint32_t l=0;l<n;l+=K)
            for(uint32_t i=min(n,l+K)-1;i>l;--i)
                suffixInBlock[i-1]=max(suffixInBlock[i],suffixInBlock[i-1]);
        uint32_t block = n / K;
        for(uint32_t i=0;i<block;++i)
            ct[0][i]=suffixInBlock[i*K];
        for(uint32_t i=1;i<bit_width(block);++i)
            memcpy(ct[i],ct[0],sizeof(uint32_t)*block);
        for (uint32_t s = 1; s < bit_width(block); ++s)
            for (uint32_t l = 0; l < block; l += (1 << (s + 1)))
            {
                for (uint32_t i = l + (1 << s) - 1; i > l; --i)
                    ct[s][i - 1] = min(ct[s][i], ct[s][i - 1]);
                for (uint32_t i = l + (1 << s), r = min(l + (1 << (s + 1)), block) - 1; i < r; ++i)
                    ct[s][i + 1] = min(ct[s][i], ct[s][i + 1]);
            }
    }
    inline uint32_t query(uint32_t l, uint32_t r)
    {
        uint32_t bl = l / K, br = r / K;
        if (bl == br)
            return *min_element(a + l, a + r + 1);
        uint32_t res = min(suffixInBlock[l], prefixInBlock[r]);
        if (bl + 1 == br)
            return res;
        ++bl, --br;
        uint32_t s = (bl == br ? 0 : bit_width(bl ^ br) - 1);
        return min({res, ct[s][bl], ct[s][br]});
    }
}
```
