$C_n=\dbinom{2n}{n}/(n+1)$

将乘除部分按照质因数存储，乘法即答案的相应质因数 +1 ，除法则 −1 ，可以优化掉高精除低精这个过程。

同时，应用高精度压位技巧，即数组中的每一位存储多个数字
```c++
typedef long long ll;
int exponential[1010];
struct big
{
    int len;
    ll x[1000];
    int bit=1e9;
    ll& operator [](int a)
    {return x[a];}
    friend void operator *=(big &a,int b)
    {
        int len=a.len,bit=a.bit;
        for(int i=1;i<=len;i++)a[i]*=b;
        int now=1;
        while(now<=len)
        {
            if(a[now]>=bit)a[now+1]+=a[now]/bit,a[now]%=bit;
            if(now==len&&a[len+1])len++;
            now++;
        }
        a.len=len;
    }
    void print()
    {
        cout<<x[len];
        for(int i=len-1;i>=1;i--)
            cout<<setw(9)<<setfill('0')<<x[i];
    }
}catalennumber;
void add(int x,int v)
{
    for(int i=2;i*i<=x;++i)
    {
        while(x%i==0)
        {
            x/=i;
            exponential[i]+=v;
        }
    }
    if(x!=1)
        exponential[x]+=v;
}
void catalan(int n)
{
    int t=n<<1;
    for(int i=n+2;i<=t;++i)
        add(i,1);
    for(int i=1;i<=n;++i)
        add(i,-1);
    catalennumber[1]=catalennumber.len=1;
    for(int i=1;i<=t;++i)
    {
        while(exponential[i])
        {
            catalennumber*=i;
            --exponential[i];
        }
    }
    catalennumber.print();
}
```
