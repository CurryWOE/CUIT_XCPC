# 最小生成树
稠密图用Prim，稀疏图用Kruskal，完全图且边权基于点权计算，用Boruvka

稠密图Prim复杂度比Kruskal更低，但不一定跑的更快
# Kruskal
每个点最初是一个连通块，每次选取边权最小的边，当边连接的是两个不同的连通块，把他们合并成一个连通块，直至生成树中有n个点

$O(E\log E)$，[洛谷模板题速度第一](https://www.luogu.com.cn/record/278627663)66ms
```cpp
const uint32_t N = 2e5 + 3, M = 2e6 + 3;
struct Kruskal
{
    struct Edge { int u, v, w; } e[M];
    int tot,n,f[N];
    int find(int x)
    {
        return f[x] == -1 ? x : (f[x] = find(f[x]));
    }
    void init(int nn)
    {
        n = nn;
        tot = 0;
        memset(f, -1, sizeof(int) * (n + 1));
    }
    void addEdge(int u, int v, int w)
    {
        e[tot++] = {u, v, w};
    }
    int run()
    {
        ranges::sort(span(e, tot),{},&Edge::w);
        int cost = 0;
        for (int i = 0;i<tot;++i)
        {
            int u=find(e[i].u),v=find(e[i].v);
            if(u==v)
                continue;
            f[u]=v;
            cost += e[i].w;
            if(--n==1)
                return cost;
        }
        return (n==1?cost:-1);
    }
};
```
## Kruskal重构树
求解一般图上对路径边权有限制的问题，使用Kruskal重构树可以把边权限制转换为二叉树上问题

边权排升序，从小到大取出一条边，如果两个端点是同一连通块则跳过，否则考虑新建节点 p，p 的点权为取出的边的边权，两个连通块的根作为 p 的左右儿子，直至只有一个连通块

性质：
1. 树的叶子都是原图节点，树的非叶点都是原图的边
2. 升序边权跑出的树，父节点点权大于左右儿子点权，降序反之
3. 升序边权跑出的树，原图两点所有路径，最大边权最小值，等于树上LCA点权，降序反之
4. 升序边权跑出的树，从原图某点出发，经过最大边权有上限，能到达的点，是树上对应点的，点权小于等于边权上限，最浅祖先子树的所有叶子，降序反之
5. 如果边权设为加边序号，还可以解决连通性问题

相较于不建新点的MST，性质3重构树时空都更优，根据[洛谷题](https://www.luogu.com.cn/problem/P1967)，[重构树](https://www.luogu.com.cn/record/279763428)152ms，[MST](https://www.luogu.com.cn/record/280249398)169ms，性质4是重构树的优点，能把满足限制的图变成子树，子树问题可以配合DFS序变成区间问题，然后就能结合各种区间数据结构
```cpp
const uint32_t N = 2e5 + 3, M = 2e6 + 3;
namespace KruskalReconstructionTree
{
    struct Edge { int u, v, w; } e[M];
    int edgeCnt,n,f[N*2];
    struct Node { int lson,rson,w; } tree[N*2];
    int find(int x)
    {
        return f[x] == 0 ? x : (f[x] = find(f[x]));
    }
    void init(int nn)
    {
        n = nn;
        edgeCnt = 0;
        memset(f, 0, sizeof(int) * (n*2));
    }
    void addEdge(int u, int v, int w)
    {
        e[edgeCnt++] = {u, v, w};
    }
    void run()
    {
        ranges::sort(span(e, edgeCnt),{},&Edge::w);
        auto idx=n;
        for(int i = 0;i<edgeCnt;++i)
        {
            int u=find(e[i].u),v=find(e[i].v);
            if(u==v)
                continue;
            tree[++idx]={u,v,e[i].w};
            f[u]=f[v]=idx;
            if(idx==2*n-1)
                break;
        }
        //重构树可能是森林
        for(auto i=n+1;i<=idx;++i)
            if(!f[i])
                //i是树根
    }
}
```
# Prim
任意点作为生成树的起点，每次选取和生成树中的点距离最小的非生成树中的点，加入生成树，直至生成树中有n个点
## 复杂度
暴力 $O(V^2+E)$，懒惰删除二叉堆 $O(E\log E)$

下为懒惰删除二叉堆板子 $O(E\log E)$，[洛谷模板题Prim速度第一](https://www.luogu.com.cn/record/278631086)86ms
```cpp
const int N = 5e3 + 3;
struct Prim
{
    struct Edge
    {
        int w,v;
        bool operator<(const Edge &f)const{return w>f.w;}
    };
    vector<Edge> e[N];
    int n,dis[N];
    bool vis[N];
    void init(int nn)
    {
        n = nn;
        for(int i=1;i<=n;++i)
            e[i].clear();
        memset(dis,0x3f,sizeof(int)*(n+1));
        memset(vis,0,sizeof(bool)*(n+1));
    }
    void addEdge(int u, int v, int w)
    {
        e[u].push_back({w,v});
        e[v].push_back({w,u});
    }
    int run(int m)
    {
        dis[1]=0;
        vector<Edge> vec{{0,1}};
        vec.reserve(m);
        priority_queue q(less<>{}, move(vec));
        int cost=0;
        while(!q.empty())
        {
            auto u=q.top().v,d=q.top().w;
            q.pop();
            if(vis[u])
                continue;
            vis[u]=true;
            cost+=d;
            if(--n==0)
                return cost;
            for(auto &[w,v]:e[u.v])
            {
                if(w>=dis[v])
                    continue;
                dis[v]=w;
                q.push({w,v});
            }
        }
        return -1;
    }
}pr;
```
# Boruvka
每次处理一个连通块时，它向其他连通块伸出一条最小边权的边，启发式合并两个连通块的点（这里总共花费 $O(nlogn)$）。如果能连成，就加上这个边的贡献，直至连通块数为1

对于边可以全部存下的图，下面提供了一种实现，$O((E+V)\log V)$。[洛谷模板题Boruvka速度第一](https://www.luogu.com.cn/record/278663396)81ms
```cpp
const uint32_t N = 5e3 + 3,M=2e5+3,INF=0x3f3f3f3f;
struct edge { uint32_t u,v,w; } e[M];
namespace BoruvkaInSparseGraph
{
    uint32_t best[N],f[N];
    uint32_t find(uint32_t x)
    {
        return f[x] == 0 ? x : (f[x] = find(f[x]));
    }
    uint32_t run(uint32_t n,uint32_t m)
    {
        uint32_t block=n,ans=0;
        e[0]={0,0,INF};
        fill(f,f+n+1,0);
        for(bool update=true;update;)
        {
            fill(best+1,best+n+1,0);
            for (uint32_t i = 1; i <= m; ++i)
            {
                auto u = find(e[i].u), v = find(e[i].v);
                if(u==v)
                    continue;
                if(e[best[u]].w>e[i].w)
                    best[u]=i;
                if(e[best[v]].w>e[i].w)
                    best[v]=i;
            }
            update = false;
            for (uint32_t i = 1; i <= n; ++i)
            {
                auto u=find(e[best[i]].u),v = find(e[best[i]].v);
                if(u==v)
                    continue;
                ans += e[best[i]].w;
                if(--block==1)
                    return ans;
                update = true;
                f[v]=u;
            }
        }
        return -1;
    }
}
```
下面介绍针对边存不下的完全图，用队列存储待合并的连通块，根据具体题目使用数学式子或者数据结构来快速求一个连通块与其他连通块的最小边权以及对应连通块，合并连通块并加上边权，直到连通块数量为一

记 $O(T)$ 是计算最小边权的复杂度，这个算法的复杂度是 $O(nT\log n)$

[CF888G异或和最小生成树Boruvka最快解法](https://codeforces.com/contest/888/submission/375636861)781ms
```cpp
int a[N];//点权，1-index
namespace BoruvkaInCompleteGraph
{
    uint32_t f[N];
    vector<uint32_t> connectBlock[N];//连通块 i 包含的点权(或点)
    queue<uint32_t> q;//待处理的连通块的根
    uint32_t find(uint32_t x)
    {
        return f[x] ==0 ? x : (f[x] = find(f[x]));
    }
    bool merge(uint32_t &u,uint32_t v)//启发式合并维护连通块包含的点
    {
        u=find(u),v=find(v);
        if(u==v)
            return false;
        if(connectBlock[u].size()<connectBlock[v].size())
            swap(u,v);
        connectBlock[u].append_range(connectBlock[v]);
        f[v]=u;
        return true;
    }
    pair<int,uint32_t> cal(int x){ return {0,0}; }//根据题目计算最小边权和对应的点
    uint64_t run(uint32_t n)
    {
        fill(f,f+n+1,0);
        for(uint32_t i=1;i<=n;++i)
        {
            connectBlock[i].clear();
            connectBlock[i].push_back(a[i]);
        }
        while(!q.empty())
            q.pop();
        for(uint32_t i=1;i<=n;++i)
            q.push(i);
        uint64_t ans=0;
        while(n>1)
        {
            while(f[q.front()])
                q.pop();
            auto u=q.front();
            q.pop();
            pair<int,uint32_t> res={INF,0};//边权，连接的另一个连通块的根
            for(auto v:connectBlock[u])
                res=min(res,cal(v));
            if(!merge(u,res[1]))
                continue;
            q.push(u);
            ans+=res[0];
            --n;
        }
        return ans;
    }
}
```
