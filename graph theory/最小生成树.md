# 最小生成树
稠密图用Prim，稀疏图用Kruskal，完全图且边权基于点权计算，用Boruvka

稠密图Prim复杂度比Kruskal更低，但不一定跑的更快
# Kruskal
每个点最初是一个连通块，每次选取边权最小的边，当边连接的是两个不同的连通块，把他们合并成一个连通块，直至生成树中有n个点

$O(E\log E)$，[洛谷模板题速度第一](https://www.luogu.com.cn/record/278627663)66ms
```cpp
const uint32_t N = 2e5 + 3, M = 2e6 + 3;
struct Kruskal
{
    struct Edge { int u, v, w; } e[M];
    int tot,n;
    int f[N];
    int find(int x)
    {
        return f[x] == -1 ? x : (f[x] = find(f[x]));
    }
    void init(int nn)
    {
        n = nn;
        tot = 0;
        memset(f, -1, sizeof(int) * (n + 1));
    }
    void addEdge(int u, int v, int w)
    {
        e[tot++] = {u, v, w};
    }
    int run()
    {
        ranges::sort(span(e, tot),{},&Edge::w);
        int cost = 0;
        for (int i = 0;i<tot;++i)
        {
            int u=find(e[i].u),v=find(e[i].v);
            if(u==v)
                continue;
            f[u]=v;
            cost += e[i].w;
            if(--n==1)
                return cost;
        }
        return (n==1?cost:-1);
    }
};
```
# Prim
任意点作为生成树的起点，每次选取和生成树中的点距离最小的非生成树中的点，加入生成树，直至生成树中有n个点

$O((V+E)\log V)$，[洛谷模板题Prim速度第一](https://www.luogu.com.cn/record/278631086)86ms
```cpp
const int N = 5e3 + 3;
struct Prim
{
    struct Edge
    {
        int w,v;
        bool operator<(const Edge &f)const{return w>f.w;}
    };
    vector<Edge> e[N],q;
    int n,dis[N];
    bool vis[N];
    void init(int nn,int m)
    {
        n = nn;
        for(int i=1;i<=n;++i)
            e[i].clear();
        memset(dis,0x3f,sizeof(int)*(n+1));
        memset(vis,0,sizeof(bool)*(n+1));
        q.clear();
        q.reserve(m);
    }
    void addEdge(int u, int v, int w)
    {
        e[u].push_back({w,v});
        e[v].push_back({w,u});
    }
    int run()
    {
        dis[1]=0;
        q.push({0,1});
        int cost=0;
        while(!q.empty())
        {
            int u=q.front().v,d=q.front().w;
            pop_heap(q.begin(),q.end());
            q.pop_back();
            if(vis[u])
                continue;
            vis[u]=true;
            cost+=d;
            if(--n==0)
                return cost;
            for(auto &[w,v]:e[u.v])
            {
                if(w>=dis[v])
                    continue;
                dis[v]=w;
                q.push_back({w,v});
                push_heap(q.begin(), q.end());
            }
        }
        return -1;
    }
}pr;
```
# Boruvka
每次处理一个连通块时，它向其他连通块伸出一条最小边权的边，启发式合并两个连通块的点（这里总共花费 $O(nlogn)$）。如果能连成，就加上这个边的贡献，直至连通块数为1

对于边可以全部存下的图，下面提供了一种实现，$O((E+V)\log V)$。[洛谷模板题Boruvka速度第一](https://www.luogu.com.cn/record/278663396)81ms
```cpp
const uint32_t N = 5e3 + 3,M=2e5+3,INF=0x3f3f3f3f;
struct edge { uint32_t u,v,w; } e[M];
namespace BoruvkaInSparseGraph
{
    uint32_t best[N],f[N];
    uint32_t find(uint32_t x)
    {
        return f[x] == 0 ? x : (f[x] = find(f[x]));
    }
    uint32_t run(uint32_t n,uint32_t m)
    {
        uint32_t block=n,ans=0;
        e[0]={0,0,INF};
        fill(f,f+n+1,0);
        for(bool update=true;update;)
        {
            fill(best+1,best+n+1,0);
            for (uint32_t i = 1; i <= m; ++i)
            {
                auto u = find(e[i].u), v = find(e[i].v);
                if(u==v)
                    continue;
                if(e[best[u]].w>e[i].w)
                    best[u]=i;
                if(e[best[v]].w>e[i].w)
                    best[v]=i;
            }
            update = false;
            for (uint32_t i = 1; i <= n; ++i)
            {
                auto u=find(e[best[i]].u),v = find(e[best[i]].v);
                if(u==v)
                    continue;
                ans += e[best[i]].w;
                if(--block==1)
                    return ans;
                update = true;
                f[v]=u;
            }
        }
        return -1;
    }
}
```
下面介绍针对边存不下的完全图，用队列存储待合并的连通块，根据具体题目使用数学式子或者数据结构来快速求一个连通块与其他连通块的最小边权以及对应连通块，合并连通块并加上边权，直到连通块数量为一

记 $O(T)$ 是计算最小边权的复杂度，这个算法的复杂度是 $O(nT\log n)$

[CF888G异或和最小生成树Boruvka最快解法](https://codeforces.com/contest/888/submission/375636861)781ms
```cpp
int a[N];//点权，1-index
namespace boruvka
{
    uint32_t f[N];
    vector<uint32_t> connectBlock[N];//连通块 i 包含的点权(或点)
    queue<uint32_t> q;//待处理的连通块的根
    uint32_t find(uint32_t x)
    {
        return f[x] ==0 ? x : (f[x] = find(f[x]));
    }
    bool merge(uint32_t u,uint32_t v)//启发式合并维护连通块包含的点
    {
        u=find(u),v=find(v);
        if(u==v)
            return false;
        if(connectBlock[u].size()<connectBlock[v].size())
            swap(u,v);
        connectBlock[u].insert(connectBlock[u].end(),connectBlock[v].begin(),connectBlock[v].end());
        f[v]=u;
        return true;
    }
    pair<int,uint32_t> cal(int x){ return {0,0}; }//根据题目计算最小边权和对应的点
    uint64_t run(uint32_t n)
    {
        fill(f,f+n+1,0);
        for(uint32_t i=1;i<=n;++i)
        {
            connectBlock[i].clear();
            connectBlock[i].push_back(a[i]);
        }
        while(!q.empty())
            q.pop();
        for(uint32_t i=1;i<=n;++i)
            q.push(i);
        uint64_t ans=0;
        while(n>1)
        {
            while(f[q.front()])
                q.pop();
            auto u=q.front();
            q.pop();
            pair<int,uint32_t> res={INF,0};//边权，连接的另一个连通块的根
            for(auto v:connectBlock[u])
                res=min(res,cal(v));
            if(!merge(u,res[1]))
                continue;
            q.push(u);
            ans+=res[0];
            --n;
        }
        return ans;
    }
}
```
