# 最短路
多起点多终点可以建源点汇点，点权变成连向源点汇点的边权，然后由汇点做最短路

dij做最短路计数不能处理边权0的情况
## 最短路径树
dijkstra，维护一个长度 n 的数组，在每个点进队列时记录这个点的入边编号，整个流程跑完后即得最短路径树的边

如果要求所有最短路径树中，边权和最小值，在遍历边时，若dis变小，则该入边记录一下，若dis相等，则记录min(之前记录的最小入边权，当前入边权)，整个流程跑完后即得“所有最短路径树中，边权和最小值”。边权和最大值反之亦然

namespace Dijkstra//O(mlog m)
{
    struct edge
    {
        int v;
        ll w;
        bool operator<(const edge &b)const{return w>b.w;}
    };
    ll dis[N];
    vector<edge> e[N],q;
    void addEdge(int u,int v,ll w)
    {
        e[u].push_back({v,w});
        // e[v].push_back({u,w});
    }
    void run(int begin)
    {
        dis[begin]=0;
        q.push_back({begin,0});
        while(!q.empty())
        {
            int u=q.front().v;
            ll d=q.front().w;
            pop_heap(q.begin(),q.end());
            q.pop_back();
            if(d!=dis[u])
                continue;
            for(auto &[v,w]:e[u])
            {
                if(dis[v]<=d+w)
                    continue;
                dis[v]=d+w;
                q.push_back({v,dis[v]});
                push_heap(q.begin(), q.end());
            }
        }
    }
    void init(int n,int m)
    {
        memset(dis,0x3f,sizeof(dis[0])*(n+1));
        for(int i=0;i<=n;++i)
            e[i].clear();
        q.reserve(m);
    }
}
namespace BellmanFordWithYenImprovement//O(nm)
{
    struct edge { int u,v,w; };
    vector<edge> forw,back;
    int dis[N];
    void init(int m)
    {
        forw.clear();
        back.clear();
        forw.reserve(m);
        back.reserve(m);
    }
    void addEdge(int u,int v,int w)
    {
        ((u < v)?forw:back).push_back({u, v, w});
    }
    bool run(int n,int begin) 
    {
        sort(forw.begin(),forw.end(),[](const edge& x, const edge& y){return x.u<y.u;});
        sort(back.begin(),back.end(),[](const edge& x, const edge& y){return x.u>y.u;});
        memset(dis, 0x3f, sizeof(int) * (n + 1));
        dis[begin] = 0;
        for (int i = n>>1; i; --i)
        {
            bool changed = false;
            for(auto* edge:{&forw, &back})
                for(auto [u,v,w]: *edge)
                    if (dis[u] != INF && dis[v] > dis[u] + w)
                    {
                        dis[v] = dis[u] + w;
                        changed = true;
                    }
            if(!changed)
                return false;
        }
        for(auto* edge:{&forw, &back})
             for (auto [u, v, w] : *edge)
                if (dis[u] != INF && dis[v] > dis[u] + w)
                    return true;
        return false;
    }
}
