# 极角排序
[洛谷速度第一](https://www.luogu.com.cn/record/282685256)
```cpp
typedef double T;
constexpr T EPS = 1e-9;
struct P
{
    T x, y;
    constexpr bool quad() const noexcept { return y > EPS || y > -EPS && x > -EPS; }
    constexpr strong_ordering operator<=>(const P& p) const noexcept
    {
        if (auto q1 = quad(), q2 = p.quad(); q1 != q2)
            return q1 <=> q2;
        auto cross = x * p.y - y * p.x;
        if (abs(cross) < EPS)
            return strong_ordering::equal;
        return cross > 0 ? strong_ordering::less : strong_ordering::greater;
    }
    constexpr bool operator==(const P &p) const noexcept { return (*this <=> p) == strong_ordering::equal; }
};
```
