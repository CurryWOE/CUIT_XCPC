# 凸包
一堆直线交点的凸包顶点，是，按斜率排序相邻直线交点集合，的子集

[[洛谷速度第二]](https://www.luogu.com.cn/record/282693606)
```cpp
typedef double T;
constexpr T EPS = 1e-9;
struct P
{
    T x, y;
    constexpr strong_ordering operator<=>(const P &p) const noexcept
    {
        if (abs(x - p.x) < EPS)
        {
            if (abs(y - p.y) < EPS)
                return strong_ordering::equal;
            return y < p.y ? strong_ordering::less : strong_ordering::greater;
        }
        return x < p.x ? strong_ordering::less : strong_ordering::greater;
    }
    constexpr bool operator==(const P &p) const noexcept { return (*this <=> p) == strong_ordering::equal; }
};
constexpr T cross(const P &p1, const P &p2, const P &p3) noexcept
{
    return (p2.x - p1.x) * (p3.y - p1.y) - (p3.x - p1.x) * (p2.y - p1.y);
}
vector<P> convexHull(vector<P> ps) noexcept
{
    if (ps.size() <= 1)
        return ps;
    ranges::sort(ps);
    uint32_t n = ps.size(),k=0;
    vector<P> qs(n * 2);
    for (uint32_t i = 0; i < n; qs[k++] = ps[i++])
        while (k > 1 && cross(qs[k - 2], qs[k - 1], ps[i]) <= EPS)
            --k;
    for (int i = n - 2, t = k; i >= 0; qs[k++] = ps[i--])
        while (k > t && cross(qs[k - 2], qs[k - 1], ps[i]) <= EPS)
            --k;
    qs.resize(k - 1);
    return qs;
}
vector<P> convexHullNonStrict(vector<P> ps) noexcept
{
    ranges::sort(ps);
    auto [first, last] = ranges::unique(ps); 
    ps.erase(first, last);
    if (ps.size() <= 1)
        return ps;
    uint32_t n = ps.size(),k=0;
    vector<P> qs(n * 2);
    for (uint32_t i = 0; i < n; qs[k++] = ps[i++])
        while (k > 1 && cross(qs[k - 2], qs[k - 1], ps[i]) < -EPS)
            --k;
    for (int i = n - 2, t = k; i >= 0; qs[k++] = ps[i--])
        while (k > t && cross(qs[k - 2], qs[k - 1], ps[i]) < -EPS)
            --k;
    qs.resize(k - 1);
    return qs;
}
```
